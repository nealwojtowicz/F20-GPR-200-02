void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    sCoord uv = fragCoord / iChannelResolution[0].xy;
    fragColor = texture(iChannel0, uv);
    sScalar lumThresh = 4.0;
    
    sScalar brightness = fragColor.r * 0.2126 + fragColor.g * 0.7152 + fragColor.b * 0.0722;
    
    if (brightness > 0.9)
        fragColor = texture(iChannel0, uv);
    else
        fragColor = texture(iChannel0, uv) * (1.0 / (1.0 + lumThresh));
}
